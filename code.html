                    function reverse(message){
                        let reversedMessage = "";

                        for (let index = message.length-1; index >=0; index--) {
                            
                            reversedMessage += message[index];
                        }
                    
                        return reversedMessage;
                    
                    }
                    
                    function displayReversed(revMessage){
                        let results = document.getElementById("results");
                        results.classList.add("revMessage");
                        results.innerHTML = revMessage;
                    }

The code is structured in a few functions:

getValue()

Gets the user input from the page.

reverse()

Takes the original string of characters and reverses it with a for loop.

displayReversed()

Displays the newly reversed string to the page
